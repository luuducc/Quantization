print("Chng trnh lng t ha xn dng sin")
f = float(input("Nhp f0: "))
N = int(input("Nhp N: "))
A = int(input("Nhp bin  A: "))
b = int(input("Nhp s bit lng t: "))
# A = 1
# f = 0.02
# N = 200

def Quantization(A,f,b,N):
    # V  th xn
    n = np.arange(0,N-1,1)
    xt = A*np.sin(2 * math.pi * f * n)
    plt.stem(n,xt)
    plt.xlabel("Trc n")
    plt.ylabel("Trc bin ")
    plt.title(" th xn")
    plt.axis([0,50, -1.1, 1.1]) 
    plt.show()
    # Hm lng t ha rounding
    def rounding(value,delta):
        result = float(value) / delta # tnh gi tr ca x/denta = result
        value = round(result) * delta # lm trn result v s nguyn gn nht, nhn li vi denta c gi tr x  lng t
        return value # tr li gi tr x  c lng t
    # V  th xqn
    xqn = []
    delta = 2*A/2**b
    for x in xt:
        xqn.append(rounding(x,delta)) # thm gi tr  c lng t ha vo mng xqn
    plt.stem(n,xqn,"r")
    plt.xlabel("Trc n")
    plt.ylabel("Trc bin ")
    plt.title(" th xqn")
    plt.axis([0,50, -1.1, 1.1])
    plt.show()
    
    # V  th eqn
    eqn = (xqn - xt)
    plt.stem(n,eqn,"y")
    plt.xlabel("Trc n")
    plt.ylabel("Trc bin ")
    plt.title(" th eqn")
    plt.axis([0,50, -delta/2+0.001, delta/2+0.001])
    plt.show()    
    
    # Tnh SQNR
    Px = A**2/2 
    eqn2 = eqn ** 2
    Pq = 0
    for x in eqn2:
        Pq += x
    Pq = Pq/N
    SQNR = Px/Pq
    SQNRlythuyet = 1.76 + 6.02 * b
    print("SQNR = ",math.log(SQNR)/math.log(10) * 10)
    print("SQNRlythuyet = ",SQNRlythuyet)
